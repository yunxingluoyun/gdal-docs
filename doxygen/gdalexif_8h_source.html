51/*
52 * TIFF Image File Directories are comprised of a table of field
53 * descriptors of the form shown below. The table is sorted in
54 * ascending order by tag. The values associated with each entry are
55 * disjoint and may appear anywhere in the file (so long as they are
56 * placed on a word boundary).
57 *
58 * If the value is 4 bytes or less, then it is placed in the offset
59 * field to save space. If the value is less than 4 bytes, it is
60 * left-justified in the offset field.
61 */
62typedef struct
63{
64 GUInt16 tdir_tag; /* see below */
65 GUInt16 tdir_type; /* data type; see below */
66 GUInt32 tdir_count; /* number of items; length in spec */
67 GUInt32 tdir_offset; /* byte offset to field data */
68} GDALEXIFTIFFDirEntry;
